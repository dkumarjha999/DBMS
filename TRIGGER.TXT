DELIMITER #
 CREATE TRIGGER T AFTER INSERT ON TRANS FOR EACH ROW 
     BEGIN
     IF NEW.TRAN_TYPE="WITHDRAWL" THEN
     UPDATE ACCOUNT SET BALANCE=BALANCE-NEW.AMOUNT WHERE ACCOUNT.ACNO=NEW.ACNO;
     ELSE IF NEW.TRAN_TYPE="DEPOSIT" THEN
     UPDATE ACCOUNT SET BALANCE=BALANCE+NEW.AMOUNT WHERE ACCOUNT.ACNO=NEW.ACNO;
     END IF;
     END IF;
     END #
